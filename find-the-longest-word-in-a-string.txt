function findLongestWordLength(str) {
  let strArray = [];
  let countArray = [];
  strArray = str.split(" ");
  for (let i = 0; i < strArray.length; i++) {
    countArray.push(strArray[i].length)
  }
  return Math.max(...countArray);
}

findLongestWordLength("The quick brown fox jumped over the lazy dog");
